	PROPERTIES

	RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin

	)

	PRIVATE include
		${CMAKE_SOURCE_DIR}/core/include
	)

	core)

	set(jsoncpp_install_dir ${INSTALL_DIR})

		jsoncpp)
	target_include_directories(quickbi-public
